You need to rationalize why that question is going to help you rule things in or out.</p>

  document.addEventListener('DOMContentLoaded', function() {
    // Set up toggle for dialogue sections
    const toggleButtons = document.querySelectorAll('.toggle-btn');
    
    toggleButtons.forEach(function(button) {
      button.addEventListener('click', function() {
        const content = this.parentNode.nextElementSibling;
        const isActive = content.classList.contains('active');
        
        // Toggle active state
        if (isActive) {
          content.classList.remove('active');
          this.classList.remove('active');
        } else {
          content.classList.add('active');
          this.classList.add('active');
        }
      });
    });
    
    // Automatically open the first dialogue section after a short delay
    setTimeout(() => {
      const firstToggle = document.querySelector('.toggle-btn');
      if (firstToggle) {
        firstToggle.click();
      }
    }, 1000);
    
    // Add page turning effect
    const pages = document.querySelectorAll('.page, .dialogue-container');
    
    function checkInView() {
      pages.forEach(function(page) {
        const rect = page.getBoundingClientRect();
        const isInView = (
          rect.top >= 0 &&
          rect.top <= (window.innerHeight || document.documentElement.clientHeight)
        );
        
        if (isInView) {
          page.classList.add('in-view');
        }
      });
    }
    
    // Initial check and add scroll listener
    checkInView();
    window.addEventListener('scroll', checkInView);
    
    // Add hover effect to annotation markers
    const markers = document.querySelectorAll('.annotation-marker');
    markers.forEach(function(marker) {
      marker.addEventListener('mouseover', function() {
        this.style.transform = 'scale(1.2)';
      });
      
      marker.addEventListener('mouseout', function() {
        this.style.transform = 'scale(1)';
      });
    });
    
    // Add page turning animations
    document.querySelectorAll('.page, .dialogue-container').forEach(function(element, index) {
      element.style.animationDelay = (index * 0.2) + 's';
      element.classList.add('animate-in');
    });
  });
</script>
